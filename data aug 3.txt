from numpy import expand_dims
from tensorflow.keras.preprocessing.image import load_img, img_to_array, ImageDataGenerator
from matplotlib import pyplot

# Load the base image
img = load_img('/content/bird.jpeg')   # <-- Replace with your path if needed
data = img_to_array(img)
samples = expand_dims(data, 0)

def show_augmented_images(datagen, title):
    it = datagen.flow(samples, batch_size=1)
    pyplot.figure(figsize=(8,8))
    for i in range(9):
        pyplot.subplot(330 + 1 + i)
        batch = next(it)
        image = batch[0].astype('uint8')
        pyplot.imshow(image)
        pyplot.axis('off')
    pyplot.suptitle(title, fontsize=16)
    pyplot.show()

# 1. Horizontal Shift
datagen = ImageDataGenerator(width_shift_range=[-200,200])
show_augmented_images(datagen, "Horizontal Shift Augmentation")

# 2. Vertical Shift
datagen = ImageDataGenerator(height_shift_range=0.5)
show_augmented_images(datagen, "Vertical Shift Augmentation")

# 3. Horizontal Flip
datagen = ImageDataGenerator(horizontal_flip=True)
show_augmented_images(datagen, "Horizontal Flip Augmentation")

# 4. Random Rotation
datagen = ImageDataGenerator(rotation_range=90)
show_augmented_images(datagen, "Random Rotation Augmentation")

# 5. Random Brightness
datagen = ImageDataGenerator(brightness_range=[0.2,1.5])
show_augmented_images(datagen, "Random Brightness Augmentation")

# 6. Random Zoom
datagen = ImageDataGenerator(zoom_range=[0.5,1.0])
show_augmented_images(datagen, "Random Zoom Augmentation")
